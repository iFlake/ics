extension installation file:
    /uname.inf
        Extension name
    /extens
        Extension files (to be copied to /extensions directory)
    /inst.inf
        Installation instructions
        Format:
            {num}: {file}
extension api:
    $mysqlic
        MySQLi connection
    $mysqlinfo
        MySQLi information
            hostname, port, username, password, database, prefix
    $registertab($name, $link)
        Register a tab in the navigation bar, returns ID
    $tabs
        Array containing all navbar tab names
            name, link
    $settabname($id, $name)
        Set a tabs name
    $settablink($id, $link)
        Set a tabs link
    $user->setpermission(name, value, [id])
        Sets a permission (removes if value is -1)
    $user->getpermission(name, [id, [ignoregroup]])
        Gets a permission
    $user->primarygroup
        Gets a user's primary group
    $user->groups
        Gets a user's groups
    group->getpermission(name, [id])
        Gets a permission
    group->setpermission(name, value, [id])
        Sets a permission

            e_forum_messages [id integer unsigned auto_increment primary key, type integer, poster integer, title text, parent integer, views integer, messages integer, created bigint unsigned, lastpost integer, content longtext]
